📋 Cómo configurar Google Sheets como backend compartido
  1. Abrí script.google.com y creá un nuevo proyecto.
  2. Borrá el código que viene por defecto y pegá el siguiente script:
var CORS = {'Access-Control-Allow-Origin':'*',
            'Access-Control-Allow-Methods':'GET,POST',
            'Access-Control-Allow-Headers':'Content-Type'};

function doOptions(e){
  return ContentService.createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT);
}

function doGet(e){
  try{
    var ss=SpreadsheetApp.getActiveSpreadsheet();
    var sh=ss.getSheetByName('9R_Records');
    var json='[]';
    if(sh&&sh.getLastRow()>=2){
      var vals=sh.getRange(2,1,sh.getLastRow()-1,11).getValues();
      var records=vals.filter(function(r){return r[3];}).map(function(r){
        return{y:r[0],m:r[1],d:r[2],clasif:r[3],area:r[4],costo:r[5],
               desc:r[6],cod:r[7],sol:r[8],motivo:r[9],cant:r[10]};
      });
      json=JSON.stringify(records);
    }
    var cb=e&&e.parameter&&e.parameter.callback;
    if(cb){
      return ContentService.createTextOutput(cb+'('+json+')')
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return ContentService.createTextOutput(json)
      .setMimeType(ContentService.MimeType.JSON);
  }catch(ex){
    var cb2=e&&e.parameter&&e.parameter.callback;
    if(cb2)return ContentService.createTextOutput(cb2+'([])')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
    return ContentService.createTextOutput('[]')
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e){
  try{
    var records=JSON.parse(e.postData.contents);
    var ss=SpreadsheetApp.getActiveSpreadsheet();
    var sh=ss.getSheetByName('9R_Records')||ss.insertSheet('9R_Records');
    sh.clearContents();
    sh.appendRow(['y','m','d','clasif','area','costo','desc','cod','sol','motivo','cant']);
    if(records&&records.length){
      sh.getRange(2,1,records.length,11).setValues(
        records.map(function(r){
          return[r.y,r.m,r.d,r.clasif||'',r.area||'',r.costo||0,
                 r.desc||'',r.cod||'',r.sol||'',r.motivo||'',r.cant||''];
        })
      );
    }
    return ContentService.createTextOutput('OK')
      .setMimeType(ContentService.MimeType.JSON);
  }catch(ex){
    return ContentService.createTextOutput('ERROR:'+ex.message)
      .setMimeType(ContentService.MimeType.JSON);
  }
}
  1. Click en Implementar → Nueva implementación.
  2. Tipo: App web. Ejecutar como: Yo. Quién tiene acceso: Cualquier usuario.
  3. Autorizá los permisos y copiá la URL que te da.
  4. Pegá esa URL en el botón ⚙ Sheets de esta página.
  5. Una vez configurado, cuando cualquiera cargue un Excel nuevo, los datos se sincronizan automáticamente para todos.
